45 static size_t
46 squash_buffer_npot_page (size_t value) {
47 #if defined(_POSIX_SOURCE)
48  static size_t page_size = 0;
49 
50  if (page_size == 0) {
51  long ps = sysconf (_SC_PAGE_SIZE);
52  page_size = (ps == -1) ? 8192 : ((size_t) ps);
53  }
54 
55  if (value < page_size)
56  value = page_size;
57 #else
58  if (value < 4096)
59  value = 4096;
60 #endif
61 
62  if ((value & (value - 1)) != 0) {
63  value -= 1;
64  value |= value >> 1;
65  value |= value >> 2;
66  value |= value >> 4;
67  value |= value >> 8;
68  value |= value >> 16;
69  value++;
70  }
71 
72  return value;
73 }

75 static void
76 squash_buffer_ensure_allocation (SquashBuffer* buffer, size_t allocation) {
77  if (allocation > buffer->allocated) {
78  allocation = squash_buffer_npot_page (allocation);
79 
80 #if defined(HAVE_MREMAP)
81  if (buffer->data == NULL) {
82  buffer->data = mmap (NULL, allocation, PROT_READ | PROT_WRITE, MAP_ANONYMOUS | MAP_PRIVATE, -1, 0);
83  buffer->allocated = allocation;
84  assert (buffer->data != MAP_FAILED);
85  } else {
86  buffer->data = mremap (buffer->data, buffer->allocated, allocation, MREMAP_MAYMOVE, NULL);
87  }
88 
89  /* lseek (buffer->fd, (off_t) buffer->allocated - 1, SEEK_SET); */
90  /* write (buffer->fd, "", 1); */
91  /* buffer->data = mmap (NULL, buffer->allocated, PROT_READ | PROT_WRITE, MAP_SHARED, buffer->fd, 0); */
92  /* assert (buffer->data != MAP_FAILED); */
93  /* fflush (stderr); */
94 #else
95  buffer->data = (uint8_t*) realloc (buffer->data, buffer->allocated);
96 #endif
97  }
98 }

107 SquashBuffer*
108 squash_buffer_new (size_t preallocated_len) {
109  SquashBuffer* buffer = (SquashBuffer*) malloc (sizeof (SquashBuffer));
110 
111 #if defined(HAVE_MREMAP)
112  buffer->data = NULL;
113  buffer->length = 0;
114  buffer->allocated = 0;
115 
116  if (preallocated_len > 0)
117  squash_buffer_ensure_allocation (buffer, preallocated_len);
118 #else
119  buffer->data = preallocated_len > 0 ? (uint8_t*) malloc (preallocated_len) : NULL;
120  buffer->length = 0;
121  buffer->allocated = preallocated_len;
122  buffer->fd = -1;
123 #endif
124 
125  return buffer;
126 }

129 squash_buffer_set_size (SquashBuffer* buffer, size_t length) {
130  if (length > buffer->allocated)
131  squash_buffer_ensure_allocation (buffer, length);

137 squash_buffer_append (SquashBuffer* buffer, uint8_t* data, size_t data_length) {
138  const size_t start_pos = buffer->length;

140  squash_buffer_set_size (buffer, buffer->length + data_length);
