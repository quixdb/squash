166 squash_options_parse_option (SquashOptions* options, const char* key, const char* value) {
167  assert (options != NULL);
168  assert (key != NULL);
169  assert (options->codec != NULL);
170 
171  const SquashOptionInfo* info = squash_codec_get_option_info (options->codec);
172 
173  if (info == NULL)
174  return SQUASH_BAD_PARAM;
175  else
176  assert (options->values != NULL);
177 
178  SquashOptionValue* val = NULL;
179  {
180  size_t option_n = 0;
181  while (info->name != NULL) {
182  if (strcasecmp (key, info->name) == 0) {
183  val = &(options->values[option_n]);
184  break;
185  }
186 
187  option_n++;
188  info++;
189  }
190  }
191 
192  if (val == NULL)
193  return SQUASH_BAD_PARAM;
194 
195  switch (info->type) {
196  case SQUASH_OPTION_TYPE_RANGE_INT:
197  case SQUASH_OPTION_TYPE_INT: {
198  int res = atoi (value);
199  if (info->type == SQUASH_OPTION_TYPE_RANGE_INT) {
200  if (!((res == 0 && info->info.range_int.allow_zero) ||
201  (res >= info->info.range_int.min && res <= info->info.range_int.max)))
202  return squash_error (SQUASH_BAD_VALUE);
203  }
204  val->int_value = res;
205  return SQUASH_OK;
206  }
207  break;
208 
209  case SQUASH_OPTION_TYPE_RANGE_SIZE:
210  case SQUASH_OPTION_TYPE_SIZE: {
211  char* endptr = NULL;
212  unsigned long long int res = strtoull (value, &endptr, 10);
213 
214  /* Parse X(KMG)[i[B]] into a size in bytes. */
215  if (*endptr != '\0') {
216  switch (*endptr++) {
217  case 'g':
218  case 'G':
219  res *= 1024;
220  case 'm':
221  case 'M':
222  res *= 1024;
223  case 'k':
224  case 'K':
225  res *= 1024;
226  break;
227  default:
228  return squash_error (SQUASH_BAD_VALUE);
229  }
230 
231  if (*endptr != '\0') {
232  if (*endptr == 'i' || *endptr == 'I')
233  endptr++;
234 
235  if (*endptr == 'b' || *endptr == 'B')
236  endptr++;
237  else
238  return squash_error (SQUASH_BAD_VALUE);
239 
240  if (*endptr != '\0')
241  return squash_error (SQUASH_BAD_VALUE);
242  }
243  }
244 
245  if (info->type == SQUASH_OPTION_TYPE_RANGE_SIZE) {
246  if (!((res == 0 && info->info.range_size.allow_zero) ||
247  (res >= info->info.range_size.min && res <= info->info.range_size.max)))
248  return squash_error (SQUASH_BAD_VALUE);
249  }
250  val->size_value = (size_t) res;
251  return SQUASH_OK;
252  }
253  break;
254 
255  case SQUASH_OPTION_TYPE_ENUM_STRING: {
256  for (unsigned int i = 0 ; info->info.enum_string.values[i].name != NULL ; i++) {
257  if (strcasecmp (value, info->info.enum_string.values[i].name) == 0) {
258  val->int_value = info->info.enum_string.values[i].value;
259  return SQUASH_OK;
260  }
261  }
262  return squash_error (SQUASH_BAD_VALUE);
263  }
264  break;
265 
266  case SQUASH_OPTION_TYPE_BOOL: {
267  if (strcasecmp (value, "true") == 0 ||
268  strcasecmp (value, "yes") == 0 ||
269  strcasecmp (value, "on") == 0 ||
270  strcasecmp (value, "t") == 0 ||
271  strcasecmp (value, "y") == 0 ||
272  strcasecmp (value, "1") == 0) {
273  val->bool_value = true;
274  } else if (strcasecmp (value, "false") == 0 ||
275  strcasecmp (value, "no") == 0 ||
276  strcasecmp (value, "off") == 0 ||
277  strcasecmp (value, "f") == 0 ||
278  strcasecmp (value, "n") == 0 ||
279  strcasecmp (value, "0") == 0) {
280  val->bool_value = false;
281  } else {
282  return squash_error (SQUASH_BAD_VALUE);
283  }
284  return SQUASH_OK;
285  }
286  break;
287 
288  case SQUASH_OPTION_TYPE_ENUM_INT: {
289  int res = atoi (value);
290  for (unsigned int i = 0 ; i < info->info.enum_int.values_length ; i++) {
291  if (res == info->info.enum_int.values[i]) {
292  val->int_value = res;
293  return SQUASH_OK;
294  }
295  }
296  return squash_error (SQUASH_BAD_VALUE);
297  }
298  break;
299 
300  case SQUASH_OPTION_TYPE_STRING:
301  val->string_value = strdup (value);
302  break;
303 
304  case SQUASH_OPTION_TYPE_NONE:
305  default:
306  squash_assert_unreachable();
307  }
308 
309  return squash_error (SQUASH_FAILED);
310 }

322  SquashStatus status = SQUASH_OK;
323  int n;
324 
325  assert (options != NULL);

328  return SQUASH_OK;

331  status = squash_options_parse_option (options, keys[n], values[n]);

470 squash_options_init (void* options,
471  SquashCodec* codec,
472  SquashDestroyNotify destroy_notify) {
473  SquashOptions* o;
474 
475  assert (options != NULL);
476  assert (codec != NULL);
477 
478  o = (SquashOptions*) options;
479 
480  squash_object_init (o, true, destroy_notify);
481  o->codec = codec;
482 
483  const SquashOptionInfo* info = squash_codec_get_option_info (codec);
484  if (info != NULL) {
485  size_t n_options;
486  for (n_options = 0 ; info[n_options].name != NULL ; n_options++) { }
487 
488  o->values = calloc (n_options, sizeof (SquashOptionValue));
489  for (size_t c_option = 0 ; c_option < n_options ; c_option++) {
490  switch (info[c_option].type) {
491  case SQUASH_OPTION_TYPE_ENUM_STRING:
492  case SQUASH_OPTION_TYPE_RANGE_INT:
493  case SQUASH_OPTION_TYPE_INT:
494  case SQUASH_OPTION_TYPE_ENUM_INT:
495  o->values[c_option].int_value = info[c_option].default_value.int_value;
496  break;
497  case SQUASH_OPTION_TYPE_BOOL:
498  o->values[c_option].bool_value = info[c_option].default_value.bool_value;
499  break;
500  case SQUASH_OPTION_TYPE_SIZE:
501  case SQUASH_OPTION_TYPE_RANGE_SIZE:
502  o->values[c_option].size_value = info[c_option].default_value.size_value;
503  break;
504  case SQUASH_OPTION_TYPE_STRING:
505  o->values[c_option].string_value = strdup (info[c_option].default_value.string_value);
506  break;
507  case SQUASH_OPTION_TYPE_NONE:
508  default:
509  squash_assert_unreachable();
510  }
511  }
512  }
513 }
